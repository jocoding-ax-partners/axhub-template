const res = await axhub.fetch("/api/v1/me", {}, {
  cookie: Astro.request.headers.get("cookie"),
});

앱 데이터 — 동적 테이블 CRUD

const ctx = { cookie: Astro.request.headers.get("cookie") };
// GET  {API_BASE}/data/{tenant}/{app}/todos
const list = await (await axhub.data("todos", {}, ctx)).json();
await axhub.data("todos", { method: "POST", body: JSON.stringify({ title: "할 일" }) }, ctx);

Astro 는 전역 쿠키 접근이 없어서 Astro.request 의 쿠키를 ctx 로 넘겨요. 헬퍼가 _hub_access 를 Bearer 로 포워딩해 사용자 자격으로 호출해요. 정적 API 키 안 써요.
